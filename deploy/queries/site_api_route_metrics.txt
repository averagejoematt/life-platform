# Site API Route Metrics — CloudWatch Logs Insights Query
# Log group: /aws/lambda/life-platform-site-api
# Time range: adjust as needed (last 24h, 7d, etc.)

# Top routes by request count + avg latency
fields @timestamp, route, duration_ms, status
| filter _type = "route_metric"
| stats count(*) as requests, avg(duration_ms) as avg_ms, max(duration_ms) as max_ms, percentile(duration_ms, 95) as p95_ms, sum(status >= 500) as errors by route
| sort requests desc
